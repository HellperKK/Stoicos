=begin
example of bind function use
=end

= :liste [1 2 3 4 5 6 7]

= :moyenne (function :tab
    {
      (bind [:length :sum]
        (Array.foldl tab [0 0]
          (function :memo :value
            {
              (bind [:l :s] memo)
              [(+ l 1) (+ s value)]
            }
          )
        )
      )
      (/. sum length)
    }
  )

println (moyenne liste)
